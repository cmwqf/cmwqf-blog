# Code

```c++
#include<cstdio>
#include<queue>
#include<cstring>
#include<iostream>
using namespace std;
const int maxN=1e5 + 100,inf=0x3f3f3f3f;
struct Node
{
    int to,next,value;
}edge[maxN*2+1];
int head[maxN+1],tot=1,n,m;
int dep[maxN+1],s,t,ans=0;
int cur[maxN+1];
int read()
{
    int num=0,f=1;
    char ch=getchar();
    while(!isdigit(ch)) {if(ch=='-') f=-1; ch=getchar();}
    while(isdigit(ch)) num=(num<<3)+(num<<1)+(ch^48),ch=getchar();
    return num*f;
}
void add(int x,int y,int t)
{
    tot++;
    edge[tot].to=y;
    edge[tot].value=t;
    edge[tot].next=head[x];
    head[x]=tot;
}
bool bfs()
{
    memset(dep,0,sizeof(dep));
    queue<int> q;
    q.push(s); dep[s]=1;
    while(!q.empty())
    {
        int x=q.front(); q.pop();
        for(int i=head[x];i;i=edge[i].next)
            if(!dep[edge[i].to]&&edge[i].value)
            {
                q.push(edge[i].to);
                dep[edge[i].to]=dep[x]+1;
                if(edge[i].to==t) return true; 
            }
    }
    return dep[t];
}
int dfs(int x,int num)
{
    if(x==t) return num;
    int sum=0;
    for(int &i=cur[x];i;i=edge[i].next)
        if(edge[i].value&&dep[edge[i].to]==dep[x]+1)
        {
            int tmp=dfs(edge[i].to,min(num,edge[i].value));
            sum+=tmp; num-=tmp;
            edge[i].value-=tmp; edge[i^1].value+=tmp;
            if(!num) break;
        }
    return sum;
}
int main()
{
    n=read(),m=read(),s=read(),t=read();
    for(int i=1;i<=m;i++)
    {
        int x=read(),y=read(),t=read();
        add(x,y,t); add(y,x,0);
    }
    while(bfs()) memcpy(cur,head,sizeof(head)),ans+=dfs(s,inf);
    printf("%d",ans);
    return 0;
}
```
